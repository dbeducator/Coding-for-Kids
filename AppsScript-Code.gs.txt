/**
 * KHANYISILE FUTURE READY LEARNERS - BACKEND
 * ============================================
 * This script turns a Google Sheet into the database for the school's
 * coding/digital-literacy programme. Deploy as a Web App (see SETUP.md).
 *
 * SHEET STRUCTURE (create these exact tab names in your Google Sheet):
 *
 * Tab "Students": StudentID | Name | Grade | Class | DateAdded
 * Tab "Progress":  Timestamp | StudentID | Module | Week | ActivityType | Score | XP | Completed
 * Tab "Badges":    Timestamp | StudentID | BadgeName
 */

function doGet(e) {
  var action = e.parameter.action;

  if (action === 'getStudents') {
    return getStudents();
  } else if (action === 'getProgress') {
    return getProgress(e.parameter.studentId);
  } else if (action === 'getClassProgress') {
    return getClassProgress(e.parameter.className);
  }

  return jsonResponse({ error: 'Unknown action' });
}

function doPost(e) {
  var action = e.parameter.action;
  var data = JSON.parse(e.postData.contents);

  if (action === 'logProgress') {
    return logProgress(data);
  } else if (action === 'addStudent') {
    return addStudent(data);
  } else if (action === 'awardBadge') {
    return awardBadge(data);
  }

  return jsonResponse({ error: 'Unknown action' });
}

// ---------- READ FUNCTIONS ----------

function getStudents() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Students');
  var rows = sheet.getDataRange().getValues();
  var students = [];

  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0]) {
      students.push({
        studentId: rows[i][0],
        name: rows[i][1],
        grade: rows[i][2],
        className: rows[i][3]
      });
    }
  }

  return jsonResponse({ students: students });
}

function getProgress(studentId) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Progress');
  var rows = sheet.getDataRange().getValues();
  var records = [];
  var totalXP = 0;

  for (var i = 1; i < rows.length; i++) {
    if (rows[i][1] === studentId) {
      var record = {
        timestamp: rows[i][0],
        studentId: rows[i][1],
        module: rows[i][2],
        week: rows[i][3],
        activityType: rows[i][4],
        score: rows[i][5],
        xp: rows[i][6],
        completed: rows[i][7]
      };
      records.push(record);
      totalXP += Number(rows[i][6]) || 0;
    }
  }

  var badgesSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Badges');
  var badgeRows = badgesSheet.getDataRange().getValues();
  var badges = [];
  for (var j = 1; j < badgeRows.length; j++) {
    if (badgeRows[j][1] === studentId) {
      badges.push(badgeRows[j][2]);
    }
  }

  return jsonResponse({ records: records, totalXP: totalXP, badges: badges });
}

function getClassProgress(className) {
  var studentsSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Students');
  var studentRows = studentsSheet.getDataRange().getValues();
  var classStudents = [];

  for (var i = 1; i < studentRows.length; i++) {
    if (studentRows[i][3] === className) {
      classStudents.push({ studentId: studentRows[i][0], name: studentRows[i][1] });
    }
  }

  var progressSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Progress');
  var progressRows = progressSheet.getDataRange().getValues();

  var summary = [];
  for (var s = 0; s < classStudents.length; s++) {
    var sid = classStudents[s].studentId;
    var completedCount = 0;
    var totalXP = 0;
    var lastActivity = '';

    for (var p = 1; p < progressRows.length; p++) {
      if (progressRows[p][1] === sid) {
        if (progressRows[p][7] === 'Y' || progressRows[p][7] === true) {
          completedCount++;
        }
        totalXP += Number(progressRows[p][6]) || 0;
        lastActivity = progressRows[p][0];
      }
    }

    summary.push({
      studentId: sid,
      name: classStudents[s].name,
      completedCount: completedCount,
      totalXP: totalXP,
      lastActivity: lastActivity
    });
  }

  return jsonResponse({ summary: summary });
}

// ---------- WRITE FUNCTIONS ----------

function logProgress(data) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Progress');
  sheet.appendRow([
    new Date(),
    data.studentId,
    data.module,
    data.week,
    data.activityType,
    data.score || '',
    data.xp || 0,
    data.completed ? 'Y' : 'N'
  ]);
  return jsonResponse({ success: true });
}

function addStudent(data) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Students');
  var studentId = data.name.replace(/\s+/g, '') + '_' + data.className.replace(/\s+/g, '');
  sheet.appendRow([studentId, data.name, data.grade, data.className, new Date()]);
  return jsonResponse({ success: true, studentId: studentId });
}

function awardBadge(data) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Badges');
  sheet.appendRow([new Date(), data.studentId, data.badgeName]);
  return jsonResponse({ success: true });
}

// ---------- HELPER ----------

function jsonResponse(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
